import java.text.SimpleDateFormat;
import java.util.Date;

public class Tarih {

	public static void main(String[] args) {
		
			Date date = new Date();
			System.out.println(date.toString());
			
			// date format örneði
			SimpleDateFormat df =
					new SimpleDateFormat("dd.MM.Y HH.mm.ss D");
			System.out.println(df.format(date));
			
			
	}

}
