import java.util.Scanner;


public class Hesap {


	public static void main(String[] args) {
		
		
		Scanner sc = new Scanner(System.in);
		
		System.out.println("Ýþlem ismi giriniz; (topla,cikar,carp,bol)");
		String islem = sc.next();
		
		System.out.println("1. sayýyý giriniz;");
		int sayi1 = sc.nextInt();
		System.out.println("2. sayýyý giriniz;");
		int sayi2 = sc.nextInt();
		int sonuc = 0;
		
		
		switch(islem){
			case "topla":
				sonuc = sayi1+sayi2;
				break;
			case "cikar":
				sonuc = sayi1-sayi2;
				break;
			case "carp":
				sonuc = sayi1*sayi2;
				break;
			case "bol":
				sonuc = sayi1/sayi2;
				break;
		}
		System.out.println("Ýþlem: "+islem+" Sayý1: "+sayi1+" Sayý2: "+sayi2+" Sonuç: "+sonuc);
		
		

	}

}
